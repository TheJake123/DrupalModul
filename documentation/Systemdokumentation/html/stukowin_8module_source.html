455 function stukowin_taxonomy_menu_submit($form, &$form_state) {
456  $sTaxonomyType = $form_state ['values'] ['type_select'] ? 'specialisation' : 'itsv';
457  $oContentManager = new content_manager ();
458  $oSourceCurr = $oContentManager->getCurriculum ( $form_state ['values'] ['curriculum_select'] );
459  $sMachineName = $oContentManager->getUniqueMachineName ( ($sTaxonomyType . '_') );
460  // Create new vocabulary
461  $aEdit = array (
462  'name' => $form_state ['values'] ['name_input'],
463  'machine_name' => $sMachineName,
464  'description' => $form_state ['values'] ['name_input'],
465  'hierarchy' => 1,
466  'module' => 'ceus_importer',
467  'weight' => 10
468  );
469  $oVocabulary = ( object ) $aEdit;
470  $oVocabulary->{'faculty'} ['und'] [0] ['value'] = $oSourceCurr ['faculty'];
471  $oVocabulary->{'version'} ['und'] [0] ['value'] = $oSourceCurr ['version'];
472  $oVocabulary->{'currtype'} ['und'] [0] ['value'] = $oSourceCurr ['type'];
473  taxonomy_vocabulary_save ( $oVocabulary );
474  // Insert structural terms
475  if (! empty ( $form_state ['values'] ['num_struc'] )) {
476  if ($form_state ['values'] ['type_select'])
477  $sText = '. Abschnitt';
478  else
479  $sText = '. Semester';
480  for($i = 1; $i <= $form_state ['values'] ['num_struc']; $i ++) {
481  $oTerm = new stdClass ();
482  $oTerm->name = $i . $sText;
483  $oTerm->vid = $oVocabulary->vid;
484  taxonomy_term_save ( $oTerm );
485  }
486  }
487  // Redirect user to taxonomy manager
488  drupal_set_message ( 'Taxonomy "' . $form_state ['values'] ['name_input'] . '" successfully created. To publish set its weight to -10' );
489  drupal_goto ( 'admin/structure/taxonomy_manager/double-tree/' . $oSourceCurr ['machine_name'] . '/' . $sMachineName );
490 }

644 function stukowin_theme_registry_alter(&$theme_registry) {

647  $theme_registry += array_diff_key ( $theme_registry_copy, $theme_registry );
648  // A list of templates the module will provide templates for
649  $hooks = array (
650  'page'
651  );
652  foreach ( $hooks as $h ) {
653  // Add the key 'theme paths' if it doesn't exist in this theme's registry
654  if (! isset ( $theme_registry [$h] ['theme paths'] ))
655  $theme_registry [$h] ['theme paths'] = array ();
656  // Shift this module's directory to the top of the theme path list
657  if (is_array ( $theme_registry [$h] ['theme paths'] )) {
658  $first_element = array_shift ( $theme_registry [$h] ['theme paths'] );
659  if ($first_element) {
660  array_unshift ( $theme_registry [$h] ['theme paths'], $first_element, drupal_get_path ( 'module', 'stukowin' ) );

662  array_unshift ( $theme_registry [$h] ['theme paths'], drupal_get_path ( 'module', 'stukowin' ) );
