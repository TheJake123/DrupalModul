443 function stukowin_taxonomy_menu_submit($form, &$form_state) {
444  $sTaxonomyType = $form_state ['values'] ['type_select'] ? 'specialisation' : 'itsv';
445  $oContentManager = new content_manager ();
446  $oSourceCurr = $oContentManager->getCurriculum ( $form_state ['values'] ['curriculum_select'] );
447  $sMachineName = $oContentManager->getUniqueMachineName ( ($sTaxonomyType . '_') );
448  // Create new vocabulary
449  $aEdit = array (
450  'name' => $form_state ['values'] ['name_input'],
451  'machine_name' => $sMachineName,
452  'description' => $form_state ['values'] ['name_input'],
453  'hierarchy' => 1,
454  'module' => 'ceus_importer',
455  'weight' => 10
456  );
457  $oVocabulary = ( object ) $aEdit;
458  $oVocabulary->{'faculty'} ['und'] [0] ['value'] = $oSourceCurr ['faculty'];
459  $oVocabulary->{'version'} ['und'] [0] ['value'] = $oSourceCurr ['version'];
460  $oVocabulary->{'currtype'} ['und'] [0] ['value'] = $oSourceCurr ['type'];
461  taxonomy_vocabulary_save ( $oVocabulary );
462  // Insert structural terms
463  if (! empty ( $form_state ['values'] ['num_struc'] )) {
464  if ($form_state ['values'] ['type_select'])
465  $sText = '. Abschnitt';
466  else
467  $sText = '. Semester';
468  for($i = 1; $i <= $form_state ['values'] ['num_struc']; $i ++) {
469  $oTerm = new stdClass ();
470  $oTerm->name = $i . $sText;
471  $oTerm->vid = $oVocabulary->vid;
472  taxonomy_term_save ( $oTerm );
473  }
474  }
475  // Redirect user to taxonomy manager
476  drupal_set_message ( 'Taxonomy "' . $form_state ['values'] ['name_input'] . '" successfully created. To publish set its weight to -10' );
477  drupal_goto ( 'admin/structure/taxonomy_manager/double-tree/' . $oSourceCurr ['machine_name'] . '/' . $sMachineName );
478 }

632 function stukowin_theme_registry_alter(&$theme_registry) {
633  $theme_registry_copy = $theme_registry;
634  _theme_process_registry ( $theme_registry_copy, 'phptemplate', 'theme_engine', 'my_custom_theme', drupal_get_path ( 'module', 'stukowin' ) );
635  $theme_registry += array_diff_key ( $theme_registry_copy, $theme_registry );
636  // A list of templates the module will provide templates for
637  $hooks = array (
638  'page'
639  );
640  foreach ( $hooks as $h ) {
641  // Add the key 'theme paths' if it doesn't exist in this theme's registry
642  if (! isset ( $theme_registry [$h] ['theme paths'] ))
643  $theme_registry [$h] ['theme paths'] = array ();
644  // Shift this module's directory to the top of the theme path list
645  if (is_array ( $theme_registry [$h] ['theme paths'] )) {
646  $first_element = array_shift ( $theme_registry [$h] ['theme paths'] );
647  if ($first_element) {
648  array_unshift ( $theme_registry [$h] ['theme paths'], $first_element, drupal_get_path ( 'module', 'stukowin' ) );
649  } else {
650  array_unshift ( $theme_registry [$h] ['theme paths'], drupal_get_path ( 'module', 'stukowin' ) );
651  }
